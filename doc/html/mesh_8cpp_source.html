134  Text/String initialization
135 -------------------------------------*/
136 bool mesh::init(const atlas& ta, const std::string& str) {
137  LS_LOG_MSG("Attempting to load a text mesh.");
138 
139  // determine the number of non-whitespace characters
140  const unsigned numChars = getDrawableCharCount(str.c_str());
141  const unsigned numVerts = numChars*MESH_VERTS_PER_GLYPH;
142  const unsigned numBytes = numVerts * sizeof(vertex);
143 
144  if (!initVertices(numVerts)) {
145  LS_LOG_ERR("\tAn error occurred while initializing a text mesh.\n");
146  return false;
147  }
148 
149  LOG_GL_ERR();
150 
151  // Attempt to get a pointer to an unsynchronized memory buffer
152  vertex* pVerts = (vertex*)vbo.mapData(
153  0, numBytes,
154  (buffer_access_t)(LS_MAP_INVALIDATE_RANGE_BIT | LS_MAP_WRITE_BIT | LS_MAP_UNSYNCHRONIZED_BIT)
155  );
156  LOG_GL_ERR();
157 
158  if (pVerts == nullptr) {
159  LS_LOG_ERR("\tUnable to send text data to the GPU through a DMA transfer.\n");
160  terminate();
161  return false;
162  }
163 
164  // Get pointers to the buffer data that will be filled with quads
165  const atlasEntry* const pGlyphs = ta.getEntries();
166  // The y-origin was found using a lot of testing. This was for resolution independence
167  float yPos = -((pGlyphs['\n'].bearing[1]*2.f)+pGlyphs['\n'].bearing[1]-pGlyphs['\n'].size[1]);
168  float xPos = 0.f;
169 
170  for (unsigned i = 0; i < str.size(); ++i) {
171  const unsigned currChar = (unsigned)str[i];
172  const atlasEntry& rGlyph = pGlyphs[currChar];
173  const float vertHang = (rGlyph.bearing[1]-rGlyph.size[1]);
174 
175  if (currChar == '\n') {
176  yPos -= (rGlyph.bearing[1]*2.f)+vertHang; // formula found through trial and error
177  xPos = 0.f;
178  }
179  else if (currChar == '\v') {
180  yPos -= ((rGlyph.bearing[1]*2.f)+vertHang)*MESH_SPACES_PER_TAB;
181  xPos = 0.f;
182  }
183  else if (currChar == '\r') {
184  xPos = 0.f;
185  }
186  else if (currChar == ' ') {
187  xPos += rGlyph.advance[0];
188  }
189  else if (currChar == '\t') {
190  xPos += rGlyph.advance[0]*MESH_SPACES_PER_TAB;
191  }
192  else {
193  const float yOffset = yPos+vertHang;
194  const float xOffset = xPos+rGlyph.bearing[0];
195  xPos += rGlyph.advance[0];
196  uploadTextGlyph(xOffset, yOffset, rGlyph, pVerts);
197  pVerts += MESH_VERTS_PER_GLYPH;
198  }
199  }
200 
201  if (!vbo.unmapData()) {
202  LS_LOG_ERR(
203  "\tAn error occurred while attempting to end a DMA transmission of text.\n"
204  );
205  vbo.unbind();
206  terminate();
207  return false;
208  }
209  LOG_GL_ERR();
210 
211  vbo.unbind();
212  numVertices = numVerts;
213  drawMode = DRAW_MODE_TRIS;
214 
215  LS_LOG_MSG("\tSuccessfully sent a string of ", numVertices, " vertices to the GPU.\n");
216 
217  return true;
218 }
