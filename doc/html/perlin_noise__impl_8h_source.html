97  Regenerate the noise permutations
98 -------------------------------------*/
99 template <typename num_t>
100 void perlinNoise_t<num_t>::seed(unsigned long s) {
101  prng->seed(s);
102 
103  // initialize all of the numbers between 0-255
104  for (int i = 0; i < 256; ++i) {
105  permutations[i] = i;
106  }
107 
108  // shuffle all of the numbers in the permutations list
109  for (unsigned i = 0; i < 256; ++i) {
110  unsigned index = (*prng)() % 256;
111  int a = permutations[i];
112  permutations[i] = permutations[index];
113  permutations[index] = a;
114  }
115 
116  // repeat all of the numbers
117  for (unsigned i = 256, j = 0; i < MAX_PERMUTATIONS; ++i, ++j) {
118  permutations[i] = permutations[j];
119  }
120 }

153  switch(hash & 0xF) {

166  case 0xC: return y + x;

174 /*-------------------------------------
175  Generate the noise function
176 
177  This implementation was as a combination of serveral sources:
178  http://flafla2.github.io/2014/08/09/perlinnoise.html
179  https://github.com/sol-prog/Perlin_Noise/blob/master/PerlinNoise.cpp
180  http://staffwww.itn.liu.se/~stegu/
181  http://staffwww.itn.liu.se/~stegu/simplexnoise/simplexnoise.pdf
182 -------------------------------------*/
183 template <typename num_t>
184 template <typename point_t>
185 num_t perlinNoise_t<num_t>::getNoise(const vec3_t<point_t>& point) {
186  // create coordinates for a "unit cube"
187  int xi = (int)std::floor(point[0]) & 255;
188  int yi = (int)std::floor(point[1]) & 255;
189  int zi = (int)std::floor(point[2]) & 255;
190 
191  double xr = point[0] - std::floor(point[0]);
192  double yr = point[1] - std::floor(point[1]);
193  double zr = point[2] - std::floor(point[2]);
194 
195  // compute how each value fades across u/v/w
196  double u = fade(xr);
197  double v = fade(yr);
198  double w = fade(zr);
199 
200  int a0 = permutations[xi ] + yi;
201  int a1 = permutations[a0 ] + zi;
202  int a2 = permutations[a0 + 1] + zi;
203  int b0 = permutations[xi + 1] + yi;
204  int b1 = permutations[b0 ] + zi;
205  int b2 = permutations[b0 + 1] + zi;
206 
207  double x10 = lerp(grad(permutations[a1], xr, yr, zr), grad(permutations[b1], xr-1, yr, zr), u);
208  double x11 = lerp(grad(permutations[a2], xr, yr-1, zr), grad(permutations[b2], xr-1, yr-1, zr), u);
209  double x12 = lerp(grad(permutations[a1+1], xr, yr, zr-1), grad(permutations[b1+1], xr-1, yr, zr-1), u);
210  double x13 = lerp(grad(permutations[a2+1], xr, yr-1, zr-1), grad(permutations[b2+1], xr-1, yr-1, zr-1), u);
211 
212  double y10 = lerp(x10, x11, v);
213  double y11 = lerp(x12, x13, v);
214 
215  return (num_t)lerp(y10, y11, w);
216 }

219  Generate the noise function with an octave (perturbations).
220 
221  This portion was found here:
222  http://flafla2.github.io/2014/08/09/perlinnoise.html
223 -------------------------------------*/
224 template <typename num_t>
225 num_t perlinNoise_t<num_t>::getOctaveNoise(const vec3_t<num_t>& point, unsigned octaves, num_t persistence) {
226  num_t total = num_t{0};
227  num_t frequency = num_t{1};
228  num_t amplitude = num_t{1};
229  num_t maxValue = num_t{1};
230 
231  for (unsigned i = 0; i < octaves; ++i) {
232  total += getNoise(point * frequency * amplitude);
233  maxValue += amplitude;
234  amplitude *= persistence;
235  frequency *= num_t{2};
236  }
237 
238  return total/maxValue;
239 }
