42 void copyGlyph(glyph& pGlyph, const FT_GlyphSlot ftGlyph) {
43  const FT_Glyph_Metrics& metrics = ftGlyph->metrics;
44  const FT_Bitmap& ftBitmap = ftGlyph->bitmap;
45 
46  // These need to be divided by 64 as their measurements are
47  // in "points," or, 1/64th of a pixel.
48  pGlyph.size = {ftBitmap.width, ftBitmap.rows};
49 
50  pGlyph.bearing = {metrics.horiBearingX, metrics.horiBearingY};
51  pGlyph.bearing /= 64;
52 
53  pGlyph.advance = {metrics.horiAdvance, metrics.vertAdvance};
54  pGlyph.advance /= 64;
55 
56  // Copy the data from FreeType into the glyph
57  std::copy(ftBitmap.buffer, ftBitmap.buffer+(ftBitmap.width*ftBitmap.rows), pGlyph.pData);
58 }

211 /*-------------------------------------*
212  Load glyphs from a FT_Face and return an array of metrics
213 
214  Most of this information was found at MBSoftworks' OpenGL tutorials.
215  http://www.mbsoftworks.sk/index.php?page=tutorials&series=1&tutorial=12
216 -------------------------------------*/
217 bool fontResource::loadGlyphs(FT_FaceRec_* ftFace) {
218  const unsigned ftNumGlyphs = (unsigned) ftFace->num_glyphs;
219  glyph* const pGlyphs = new(std::nothrow) glyph[ftNumGlyphs];
220 
221  // Dynamic array error
222  if (pGlyphs == nullptr) {
223  return false;
224  }
225 
226  for (unsigned i = 0; i < ftNumGlyphs; ++i) {
227  FT_Error ftErr;
228  unsigned charIndex = FT_Get_Char_Index(ftFace, i);
229 
230  ftErr = FT_Load_Glyph(ftFace, charIndex, FT_LOAD_RENDER);
231 
232  if (ftErr) {
233  LS_LOG_ERR(
234  "\tUnable to load a glyph at index ", charIndex,
235  "\n\tFunction: FT_Load_Glyph",
236  "\n\tError Code: ", FT_Errors[ftErr].code,
237  "\n\tFT Error: ", FT_Errors[ftErr].message
238  );
239  delete [] pGlyphs;
240  return false;
241  }
242 
243  ftErr = FT_Render_Glyph(ftFace->glyph, FT_RENDER_MODE_NORMAL);
244  if (ftErr) {
245  LS_LOG_ERR(
246  "\tUnable to render the glyph at index ", charIndex,
247  "\n\tFunction: FT_Render_Glyph",
248  "\n\tError Code: ", FT_Errors[ftErr].code,
249  "\n\tFT Error: ", FT_Errors[ftErr].message
250  );
251  delete [] pGlyphs;
252  return false;
253  }
254 
255  const FT_GlyphSlot ftGlyph = ftFace->glyph;
256  const FT_Bitmap* ftBitmap = &ftGlyph->bitmap;
257  const math::vec2i bmpSize = {ftBitmap->width, ftBitmap->rows};
258  const int byteSize = bmpSize[0]*bmpSize[1];
259  char* const data = new (std::nothrow) char[byteSize];
260 
261  if (data == nullptr) {
262  LS_LOG_ERR("\tUnable to allocate ", byteSize, " bytes for glyph data.");
263  dataSize = 0;
264  maxGlyphSize = math::vec2i{0,0};
265  delete [] pGlyphs;
266  return false;
267  }
268 
269  pGlyphs[i].pData = data;
270 
271  copyGlyph(pGlyphs[i], ftGlyph);
272 
273  dataSize += byteSize;
274 
275  maxGlyphSize[0] = math::max(maxGlyphSize[0], bmpSize[0]);
276  maxGlyphSize[1] = math::max(maxGlyphSize[1], bmpSize[1]);
277  }
278 
279  pData = reinterpret_cast<char*>(pGlyphs);
280  numGlyphs = ftNumGlyphs;
281 
282  return true;
283 }
