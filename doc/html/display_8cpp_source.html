90 bool display::init(void* const hwnd) {
91  LS_LOG_MSG("Attempting to link with a preexisting display.");
92 
93  if (hwnd == nullptr) {
94  LS_LOG_ERR("\tNo display information provided.\n");
95  return false;
96  }
97 
98  LS_LOG_MSG("\tLoading internal flags for a native window.");
99 
100 
101  display tempDisp{};
102 
103  if (!tempDisp.init(math::vec2i{0})) {
104  LS_LOG_ERR("\tUnable to create a shared context for a native window.\n");
105  return false;
106  }
107  else {
108  printWindowFlags(SDL_GetWindowFlags(tempDisp.getWindow()));
109 
110  // SDL requires a string, containing the SDL_Window* pointer, to be
111  // formatted with "%p". using std::to_string() does not work.
112  char nativeHandleStr[sizeof(void*)*2]; // <--- only big enough to stringify a pointer
113  sprintf(nativeHandleStr, "%p", (void*)tempDisp.getWindow());
114 
115  LS_LOG_MSG("\tCopying window flags from ", nativeHandleStr, " to ", hwnd, '.');
116 
117  const int wasFlagCopied = SDL_SetHintWithPriority(
118  SDL_HINT_VIDEO_WINDOW_SHARE_PIXEL_FORMAT,
119  nativeHandleStr, SDL_HINT_OVERRIDE
120  );
121 
122  if (wasFlagCopied != SDL_TRUE) {
123  LS_LOG_ERR("\tUnable to toggle internal OpenGL 3.3 compatibility.\n");
124  return false;
125  }
126  }
127 
128  // Create the window
129  pWindow = SDL_CreateWindowFrom(hwnd);
130 
131  if (!pWindow) {
132  LS_LOG_ERR("\tUnable to link with a preexisting display.");
133  LS_LOG_ERR('\t', SDL_GetError(), '\n');
134  terminate();
135  return false;
136  }
137  else {
138  LS_LOG_MSG("\tSuccessfully linked with a preexisting display.\n");
139  windowIsNative = true;
140  printWindowFlags(SDL_GetWindowFlags(pWindow));
141  SDL_GL_LoadLibrary(nullptr); // must be called in order to avoid crashing
142  }
143 
144  return true;
145 }

168 
169  // Create the window
170  pWindow = SDL_CreateWindow(

172  inResolution[0], inResolution[1], windowFlags
173  );
174 
175  if (!pWindow) {
176  LS_LOG_ERR("\tUnable to create a display object.");
177  LS_LOG_ERR('\t', SDL_GetError(), '\n');
178  terminate();
179  return false;
180  }
181  else {
182  windowIsNative = false;
183  LS_LOG_MSG("\tSuccessfully created a window through SDL.");
184  }
