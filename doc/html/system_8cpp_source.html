132 /*-------------------------------------
133  SubSystem Running
134 -------------------------------------*/
135 void system::run() {
136  if (!gameList.size()) {
137  LS_LOG_GAME_ERR("No game states are available!\n", SDL_GetError(), '\n');
138  this->stop();
139  return;
140  }
141 
142  // Ensure the display is still open
143 #ifdef LS_DEBUG
144  if (!pDisplay || !pDisplay->isRunning()) {
145  this->stop();
146  return;
147  }
148 #endif
149 
150  SDL_Event pEvent;
151  while (SDL_PollEvent(&pEvent)) {
152  // Hardware events passed through SDL
153  for (unsigned i = 0; i < gameList.size(); ++i) {
154  passHardwareEvents(pEvent, gameList[i]);
155  }
156  }
157 
158  // Frame Time Management
159  const float currTime = (float)SDL_GetTicks(); // SDL uses millisecond timing.
160  tickTime = currTime-prevTime;
161  prevTime = currTime;
162 
163  // Game state management
164  updateGameStates(tickTime);
165 
166  // Render to the screen after all events have been processed
167  ls::draw::display& disp = *pDisplay;
168  renderContext.makeCurrent(disp);
169  renderContext.flip(disp);
170  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
171 }

210 /*-------------------------------------
211  Update the states in the game stack
212 -------------------------------------*/
213 void system::updateGameStates(float tickTime) {
214  for(unsigned i = 0; i < gameList.size(); ++i) {
215  switch(gameList[i]->getState()) {
216 
217  case LS_GAME_RUNNING:
218  gameList[i]->onRun(tickTime);
219  break;
220 
221  case LS_GAME_PAUSED:
222  gameList[i]->onPause(tickTime);
223  break;
224 
225  case LS_GAME_STOPPED:
226  popGameState(i);
227  i -= 1;
228  break;
229 
230  case LS_GAME_INIT:
231  if (gameList[i]->onStart() == true) {
232  gameList[i]->setState(LS_GAME_RUNNING);
233  }
234  else {
235  LS_LOG_GAME_ERR("ERROR: A new gameState was unable to start.");
236  gameList[i]->setState(LS_GAME_STOPPED);
237  }
238 
239  default:
240  break;
241  }
242  }
243 }
