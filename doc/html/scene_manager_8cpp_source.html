119 /*
120  * Initialization of the default gray and magic pink textures
121  */
122 bool sceneManager::initDefaultTexture() {
123  if (defaultTex.getId()) {
124  return true;
125  }
126 
127  if (!defaultTex.init(
128  0, LS_RGBA_32F,
129  math::vec2i{2},
130  LS_RGBA, COLOR_TYPE_FLOAT,
131  (void*)checkeredCol)
132  ) {
133  LS_LOG_ERR("\tUnable to initialize a default texture for draw models.");
134  LOG_GL_ERR();
135 
136  return false;
137  }
138  else {
139  defaultTex.setParameter(LS_TEX_MIN_FILTER, LS_FILTER_NEAREST);
140  defaultTex.setParameter(LS_TEX_MAG_FILTER, LS_FILTER_NEAREST);
141  defaultTex.setParameter(LS_TEX_WRAP_S, LS_TEX_REPEAT);
142  defaultTex.setParameter(LS_TEX_WRAP_T, LS_TEX_REPEAT);
143  LOG_GL_ERR();
144  }
145 
146  return true;
147 }
