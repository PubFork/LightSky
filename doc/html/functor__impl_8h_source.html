51  Functor parameter type-checking for a single argument.
52 -------------------------------------*/
53 template <typename arg_t>
54 bool functor::checkSingleArg(const functor& f, unsigned i, arg_t* t) {
55  if (i >= f.getNumArgs()) {
56  std::cerr
57  << "Error: Incorrect number of arguments being compiled: "
58  << "\n\tFunction: " << (void*)&f << '-' << f.getScriptSubType()
59  << "\n\tParameter: " << i << '/' << f.getNumArgs()
60  << std::endl;
61  return false;
62  }
63 
64  if (f.getArg(i) == nullptr) {
65  std::cerr
66  << "Error: Function contains a null parameter: "
67  << "\n\tFunction: " << (void*)&f << '-' << f.getScriptSubType()
68  << "\n\tParameter: " << i << '/' << f.getNumArgs()
69  << std::endl;
70  return false;
71  }
72 
73  if (dynamic_cast<const arg_t*>(f.getArg(i)) == nullptr) {
74  std::cerr
75  << "Error: TypeID of function argument does not match the type " \
76  " declared in its parent function."
77  << "\n\tFunction: " << (void*)&f << '-' << f.getScriptSubType()
78  << "\n\tInput type: " << typeid(f.getArg(i)).name()
79  << "\n\tActual type: " << typeid(t).name()
80  << std::endl;
81  return false;
82  }
83  return true;
84 }
