126 bool meshModel::init(const mesh& m, const texture& t) {
127  // there's no telling what data might be pushed into the VAO. Get rid of it.
128  vao.terminate();
129  if (!vao.init()) {
130  LS_LOG_ERR("Unable to create a drawModel for mesh ", m.getId(), '.');
131  terminate();
132  return false;
133  }
134 
135  // vbos are resized when any new data is pushed into them
136  if (modelVbo.isValid() == false) {
137  if (!modelVbo.init()) {
138  LS_LOG_ERR("\tUnable to initialize a model matrix buffer.");
139  terminate();
140  return false;
141  }
142  else {
143  // ensure at lease one model matrix is available on initialization.
144  const math::mat4 identityMat = {1.f};
145  setNumInstances(1, &identityMat);
146  }
147  }
148 
149  pMesh = &m;
150 
151  setVertexAttribs();
152  setTexture(t);
153 
154  return true;
155 }
