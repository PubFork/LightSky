91 template <typename num_t> inline
92 math::quat_t<num_t> math::slerp(const quat_t<num_t>& q1, const quat_t<num_t>& q2, num_t t) {
93  num_t dotProd = dot(q1, q2);
94 
95  // Reverse the sign of q2 if q1.q2 < 0.
96  if (dotProd < num_t{0}) {
97  dotProd = -dotProd;
98  }
99 
100  const num_t theta = acos(dotProd);
101  num_t mult1, mult2;
102 
103  if (theta >= LS_EPSILON) {
104  const num_t st = (num_t) LS_SIN(theta);
105  mult1 = LS_SIN((num_t{1} - t) * theta) / st;
106  mult2 = LS_SIN(t * theta) / st;
107  }
108  else {
109  // To avoid division by 0 and by very small numbers the approximation of sin(angle)
110  // by angle is used when theta is small (0.000001 is chosen arbitrarily).
111  mult1 = num_t{1} - t;
112  mult2 = t;
113  }
114 
115  return quat_t<num_t>{
116  mult1 * q1.q[0] + mult2 * q2.q[0],
117  mult1 * q1.q[1] + mult2 * q2.q[1],
118  mult1 * q1.q[2] + mult2 * q2.q[2],
119  mult1 * q1.q[3] + mult2 * q2.q[3]
120  };
121 }

187  matToQuat
188 -------------------------------------*/
189 template <typename num_t> inline
190 math::quat_t<num_t> math::matToQuat(const mat3_t<num_t>& m) {
191  num_t s(0);
192  num_t trace(m.m[0][0] + m.m[1][1] + m.m[2][2]);
193  quat_t<num_t> q;
194 
195  if (trace > num_t{0}) {
196  s = LS_SQRT(trace + num_t{1});
197  q.q[3] = s * num_t{0.5};
198  s = num_t{0.5} / s;
199  q.q[0] = (m.m[1][2] - m.m[2][1]) * s;
200  q.q[1] = (m.m[2][0] - m.m[0][2]) * s;
201  q.q[2] = (m.m[0][1] - m.m[1][0]) * s;
202  }
203  else {
204  int nxt[3] = {1, 2, 0};
205  int i(0), j(0), k(0);
206 
207  if (m.m[1][1] > m.m[0][0]) {
208  i = 1;
209  }
210 
211  if (m.m[2][2] > m.m[i][i]) {
212  i = 2;
213  }
214 
215  j = nxt[i];
216  k = nxt[j];
217  s = LS_SQRT((m.m[i][i] - (m.m[j][j] + m.m[k][k])) + num_t{1});
218 
219  q[i] = s * num_t{0.5};
220  s = num_t{0.5} / s;
221  q[3] = (m.m[j][k] - m.m[k][j]) * s;
222  q[j] = (m.m[i][j] + m.m[j][i]) * s;
223  q[k] = (m.m[i][k] + m.m[k][i]) * s;
224  }
225 
226  return q;
227 }

325  toEuler
326 -------------------------------------*/
327 template <typename num_t> inline
328 math::vec3_t<num_t> math::toEuler(const quat_t<num_t>& q) {
329  const num_t xx = q.q[0] * q.q[0];
330  const num_t yy = q.q[1] * q.q[1];
331  const num_t zz = q.q[2] * q.q[2];
332 
333  const num_t pitch = std::atan2(
334  num_t{2} * ((q.q[3] * q.q[0]) * (q.q[1] * q.q[2])),
335  num_t{1} - (num_t{2} * (xx + yy))
336  );
337  const num_t yaw = std::asin(num_t{2} * ((q.q[3] * q.q[1]) - (q.q[2] * q.q[0])));
338  const num_t roll = std::atan2(
339  num_t{2} * ((q.q[3] * q.q[2]) + (q.q[0] * q.q[1])),
340  num_t{1} - (num_t{2} * (yy + zz))
341  );
342 
343  return vec3_t<num_t>{pitch, yaw, roll};
344 }

349 template <typename num_t> inline
350 math::quat_t<num_t> math::fromEuler(const math::vec3_t<num_t>& angles) {
351  const num_t cp = LS_COS(angles.v[0] * num_t{0.5});
352  const num_t sp = LS_SIN(angles.v[0] * num_t{0.5});
353 
354  const num_t cy = LS_COS(angles.v[1] * num_t{0.5});
355  const num_t sy = LS_SIN(angles.v[1] * num_t{0.5});
356 
357  const num_t cr = LS_COS(angles.v[2] * num_t{0.5});
358  const num_t sr = LS_SIN(angles.v[2] * num_t{0.5});
359 
360  return math::quat_t<num_t>{
361  (sy * cp * cr) - (cy * sp * sr),
362  (cy * sp * cr) + (sy * cp * sr),
363  (cy * cp * sr) - (sy * sp * cr),
364  (cy * cp * cr) + (sy * sp * sr)
365  };
366 }
