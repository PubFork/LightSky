72  fastInvSqrt

74  The fast inverse square root method adopted for regular square rooting.
75  this method was found at:
76  http://rrrola.wz.cz/inv_sqrt.html

123 /*-------------------------------------
124  fastLog2
125 
126  Fast Approximate logarithms
127  This method was found on flipcode:
128  http://www.flipcode.com/archives/Fast_log_Function.shtml
129  Accurate to within 5 decimal places.
130  This method relies on the IEEE floating point specification
131 -------------------------------------*/
132 template <> inline
133 float math::fastLog2<float>(float n) {
134  int* const exp = reinterpret_cast<int*> (&n);
135  int x = *exp;
136 
137  const int log2 = ((x >> 23) & 255) - 128;
138 
139  x &= ~(255 << 23);
140  x += 127 << 23;
141 
142  *exp = x;
143  n = ((-1.f / 3.f) * n + 2.f) * n - 2.f / 3.f;
144  return n + log2;
145 }
